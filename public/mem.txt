  <script>
    async function loadMemory() {
      const res = await fetch('/memory');
      const data = await res.json();

      const memoryList = document.getElementById('memoryList');
      memoryList.innerHTML = '';

      data.forEach(item => {
        const div = document.createElement('div');
        div.className = 'memory-item';
        div.innerHTML = `
          <span>${item.text}</span>
          <button class="delete" onclick="deleteMemory('${item.id}')">üóëÔ∏è</button>
        `;
        memoryList.appendChild(div);
      });
    }

    async function addMemory() {
      const text = document.getElementById('memoryInput').value.trim();
      if (!text) return;

      await fetch('/memory', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ text })
      });

      document.getElementById('memoryInput').value = '';
      loadMemory();
    }

    async function deleteMemory(id) {
      await fetch(`/memory/${id}`, { method: 'DELETE' });
      loadMemory();
    }

    loadMemory();

    function checkPassword() {
        const input = document.getElementById('passwordInput').value;
        const correct = 'card432'; // üëâ change this to your real password

        if (input === correct) {
            document.getElementById('lockScreen').style.display = 'none';
        } else {
            alert('Incorrect password üò¨');
        }
    }
